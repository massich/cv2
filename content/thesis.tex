
\section{PhD Thesis}
\cvitem{title}{\emph{Deformable object segmentation in ultra-sound images.}}
\cvitem{supervisors}{\href{http://eia.udg.edu/~joanm/index-uk.html}{Joan Mart\'i} and \href{http://scholar.google.fr/citations?user=tNttgvEAAAAJ&hl=en&oi=sra}{Prof. Fabrice Meriaudeau}}
\cvitem{description}{This thesis is devoted to automatic segmentation of breast lesions in ultrasound images, since this task is key for the development of robust Computer Aided Diagnosis (CAD) systems applied to this organ and image modality. The proposed segmentation strategy divides the images into meaningful regions called super-pixels and labels them using a minimization framework that takes into account training and regularization.}
